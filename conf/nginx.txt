Directory:
/etc/nginx/sites-available/wiki

Config file:
server {
    listen 8001;
    server_name localhost;

    root /home/daniswastaken/wiki/docs/.vitepress/dist;
    index index.html;

    location / {
        # Added $uri.html here
        try_files $uri $uri.html $uri/ =404;
    }

    # Highly recommended for VitePress
    error_page 404 /404.html;
}

Enable the nginx conf:
sudo ln -s /etc/nginx/sites-available/wiki /etc/nginx/sites-enabled/
sudo nginx -t
